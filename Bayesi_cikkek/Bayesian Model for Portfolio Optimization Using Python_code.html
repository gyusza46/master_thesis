# This code uses a probabilistic approach to optimize the portfolio by sampling the covariance matrix from a Wishart distribution, which is 
# often used to model the distribution of sample covariance matrices. It then maximizes the Sharpe ratio, which is a measure of the return 
# relative to risk.

# We calculate the returns, mean_returns, and cov_matrix. Then, we define the prior distributions, which is a normal distribution is used to 
# model the prior belief about the expected returns of the assets, with a mean equal to the average returns of the assets and a small 
# standard deviation of 0.1.

# The custom function wishart_sample generates a sample covariance matrix from a Wishart distribution. This simulates the uncertainty in the 
# covariance matrix (risk of the assets) based on the degrees of freedom (df=num_assets) and a scale matrix (identity matrix in this case).

# The objective function calculates the Sharpe ratio, which is the portfolio's return divided by its standard deviation (risk). The goal is

import scipy.optimize as opt

# Objective function: minimize portfolio risk (variance), while maximizing return
def objective(weights, mean_returns, cov_matrix):
    portfolio_return = np.dot(weights, mean_returns)
    portfolio_variance = np.dot(weights.T, np.dot(cov_matrix, weights))
    portfolio_risk = np.sqrt(portfolio_variance)
    # Negative return since we minimize risk, but want to maximize return
    return portfolio_risk - portfolio_return  # We minimize the risk while maximizing return

# Sum of weights must be 1 (fully invested portfolio)
constraints = ({'type': 'eq', 'fun': lambda weights: np.sum(weights) - 1})

# Bounds for each weight: between 0 and 1
bounds = tuple((0, 1) for asset in range(len(mean_returns)))

# Initial guess for the weights (equal allocation)
initial_weights = np.ones(len(mean_returns)) / len(mean_returns)


# BEFORE
# Portfolio Return (Before Optimization)
portfolio_return_before = np.dot(initial_weights, mean_returns)

# Portfolio Risk (Before Optimization) - Variance
portfolio_variance_before = np.dot(initial_weights.T, np.dot(cov_matrix, initial_weights))
portfolio_risk_before = np.sqrt(portfolio_variance_before)  # Standard deviation (sqrt of variance)

print(f"Before Optimization: Portfolio Return = {portfolio_return_before:.4f}")
print(f"Before Optimization: Portfolio Risk = {portfolio_risk_before:.4f}")


# AFTER
# Optimize portfolio weights
result = opt.minimize(objective, initial_weights, args=(mean_returns, cov_matrix), method='SLSQP', bounds=bounds, constraints=constraints)

# Extract the optimized weights
optimized_weights = result.x

# Portfolio Return (After Optimization)
portfolio_return_after = np.dot(optimized_weights, mean_returns)

# Portfolio Risk (After Optimization) - Variance

# Bayesian Model Applications

# Prior Beliefs:
# These are your initial assumptions about the parameters (e.g., expected returns, covariances of asset returns) before seeing the data.
    
# For example, we might assume that the expected returns for each asset follow a certain distribution, and the covariance matrix might be 
# based on historical correlations.

# Posterior:
# After observing the data (new evidence), we update our beliefs. The posterior distribution gives us the updated probabilities of the 
# parameters after taking the data into account.
    
# This is the key output of Bayesian updating. The posterior distribution becomes the new belief about the parameters and can be used for 
# decision-making moving forward.

# We consider the following:
# Dynamic updating: When new data is available (e.g., updated asset returns), the Bayesian model updates the expected returns and covariance 
# estimates accordingly. 

# Incorporating uncertainty: Rather than relying solely on point estimates (like the mean of returns), the Bayesian model also incorporates the 
# uncertainty about these estimates. This leads to a more robust portfolio because it considers potential variations in expected returns and risks.

# Improved decisions: Bayesian methods give you a more refined and probabilistic approach, allowing for better decision-making over time 
# as new information becomes available.
